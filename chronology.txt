//html collection
spanElements = messages[i].getElementsByTagName('span');

for(var j = 0; j < spanElements.length; j++){
var hasLink = spanElements[j].getElementsByTagName('a');

if(!hasLink || hasLink[0] === undefined){
continue;

var url;
